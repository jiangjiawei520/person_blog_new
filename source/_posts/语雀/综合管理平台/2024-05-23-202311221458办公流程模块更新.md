## 新增数据库表portal_gd库：
```plsql
CREATE TABLE `zgpt_meeting` (
  `id` varchar(36) NOT NULL COMMENT '主键id',
<!--more-->
  `name` varchar(255) DEFAULT NULL COMMENT '会议名称',
  `place` varchar(255) DEFAULT NULL COMMENT '会议地点',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_time` datetime DEFAULT NULL COMMENT '修改时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=gbk COMMENT='会议表;'

CREATE TABLE `zgpt_leave_apply` (
  `id` varchar(36) NOT NULL COMMENT '主键id',
  `leave_type` varchar(3) DEFAULT NULL COMMENT '请假类型（字典编码）',
  `start_time` datetime DEFAULT NULL COMMENT '开始时间',
  `end_time` datetime DEFAULT NULL COMMENT '结束时间',
  `reason` varchar(1000) DEFAULT NULL COMMENT '请假事由',
  `file_id` varchar(500) DEFAULT NULL COMMENT '附件id（多个用逗号,分隔开）',
  `user_id` varchar(36) DEFAULT NULL COMMENT '用户id',
  `dept_id` varchar(36) DEFAULT NULL COMMENT '部门id',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `user_id_idx` (`user_id`) USING BTREE,
  KEY `dept_id_idx` (`dept_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=gbk COMMENT='请假申请表;'

CREATE TABLE `zgpt_meeting_apply` (
  `id` varchar(36) NOT NULL COMMENT '主键id',
  `meet_id` varchar(36) DEFAULT NULL COMMENT '会议id',
  `theme` varchar(255) DEFAULT NULL COMMENT '会议主题',
  `today` datetime DEFAULT NULL COMMENT '当天日期',
  `start_time` datetime DEFAULT NULL COMMENT '会议开始时间',
  `end_time` datetime DEFAULT NULL COMMENT '会议结束时间',
  `attendance` int(3) DEFAULT NULL COMMENT '参会人数',
  `participant` varchar(1000) DEFAULT NULL COMMENT '参会人员',
  `file_id` varchar(500) DEFAULT NULL COMMENT '附件id（多个用逗号,分隔开）',
  `user_id` varchar(36) DEFAULT NULL COMMENT '申请用户id',
  `dept_id` varchar(36) DEFAULT NULL COMMENT '部门id',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `meet_id_idx` (`meet_id`) USING BTREE,
  KEY `user_id_idx` (`user_id`) USING BTREE,
  KEY `dept_id_idx` (`dept_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=gbk COMMENT='会议申请表;'

CREATE TABLE `zgpt_outing_apply` (
  `id` varchar(36) NOT NULL COMMENT '主键id',
  `reason` varchar(1000) DEFAULT NULL COMMENT '外出事由',
  `start_time` datetime DEFAULT NULL COMMENT '开始时间',
  `end_time` datetime DEFAULT NULL COMMENT '结束时间',
  `file_id` varchar(500) DEFAULT NULL COMMENT '附件id（多个用逗号,分隔开）',
  `user_id` varchar(36) DEFAULT NULL COMMENT '用户id',
  `dept_id` varchar(36) DEFAULT NULL COMMENT '部门id',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `user_id_idx` (`user_id`) USING BTREE,
  KEY `dept_id_idx` (`dept_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=gbk COMMENT='外出申请表;'

CREATE TABLE `zgpt_seal_apply` (
  `id` varchar(36) NOT NULL COMMENT '主键id',
  `reason` varchar(1000) DEFAULT NULL COMMENT '用印事由',
  `seal_type` varchar(3) DEFAULT NULL COMMENT '印章类型（字典编码）',
  `document_name` varchar(255) DEFAULT NULL COMMENT '文档名称',
  `document_num` int(3) DEFAULT NULL COMMENT '文档份数',
  `seal_time` datetime DEFAULT NULL COMMENT '用印日期',
  `file_id` varchar(500) DEFAULT NULL COMMENT '附件id（多个用逗号,分隔开）',
  `user_id` varchar(36) DEFAULT NULL COMMENT '用户id',
  `dept_id` varchar(36) DEFAULT NULL COMMENT '部门id',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `user_id_idx` (`user_id`) USING BTREE,
  KEY `dept_id_idx` (`dept_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=gbk COMMENT='用章申请表;'
```
