SELECT
               sqb.*,
               m.yflx,
               m.yfxq,
               m.qyzb,
               f.fdfs_path report_file_path
        FROM apply_kczy_yfycx_sqb sqb
                LEFT JOIN apply_kczy_yfycx m on sqb.ywid=m.ywid
                LEFT JOIN fdfs_record f on sqb.report_file_id=f.id
                INNER JOIN apply_process_info a ON sqb.ywid = a.ywid
        WHERE sqb.rec_flag = '0'
          AND a.sjzt = '2'
