## 当前环节（xmcx_DQHJ）
```sql
create or replace view xmcx_dqhj as
select one_process.processinstid, ywid, processdefname,
       case when subflow.parentprocid is not null and processdefname='com.geostar.gtgh.jsydsp.workflow.define.three.stsp' then '省厅会审'
            when ainst.activityinstname is null and pinst.currentstate = 7 then '已完成'
            else ainst.activityinstname  end activityinstname
<!--more-->
from(
    select extend2 ywid, max(processinstid) processinstid from tdglbps.wfprocessinst
    where processdefname in (
              'com.geostar.gtgh.jsydsp.workflow.define.three.xjsp',
              'com.geostar.gtgh.jsydsp.workflow.define.three.sjsp',
              'com.geostar.gtgh.jsydsp.workflow.define.three.stsp',
              'com.geostar.gtgh.jsydsp.workflow.define.three.xjbl',
              'com.geostar.gtgh.jsydsp.workflow.define.three.sjbl',
              'com.geostar.gtgh.jsydsp.workflow.define.three.sjba',
              'com.geostar.gtgh.jsydsp.workflow.define.three.stba',
              'com.geostar.gtgh.jsydsp.workflow.define.three.stsp_bsfpf'
          )
  	and  PROCESSINSTID not in (select PROCESSINSTID from tdglbps.wftransition t where t.DESTACTDEFID='finishActivity' and t.SRCACTDEFID='shj_lycsh')
    group by extend2
)one_process
left join tdglbps.wfprocessinst pinst
on one_process.processinstid = pinst.processinstid
left join (
     select max(activityinstid) activityinstid, processinstid from tdglbps.wfactivityinst
     where currentstate in (2,3) and activitytype != 'subflow'
     group by processinstid
) acti
on pinst.processinstid = acti.processinstid
left join tdglbps.wfactivityinst ainst
on acti.activityinstid = ainst.activityinstid
left join (
    select parentprocid from tdglbps.wfprocessinst
    where currentstate = 2
    group by parentprocid
) subflow
on pinst.processinstid = subflow.parentprocid;
```

## 新案卷-在办件统计
```sql
SELECT
	xzqhdm,
	hjmc,
	sum( CASE WHEN ywlx = 'PCBP' THEN 1 ELSE 0 END ) pcbpsl,
	sum( CASE WHEN ywlx = 'DDXZ' THEN 1 ELSE 0 END ) ddxzsl,
	sum( CASE WHEN ywlx = 'SSFA' THEN 1 ELSE 0 END ) ssfasl 
FROM
	(
	SELECT
		t.processinstid,
		t.extend2 ywid,
		t.extend1 xmmc,
	CASE
			
			WHEN extend9 IN ( 'SJZJ', 'SJBL' ) THEN
			substr( t.extend5, 0, 4 ) || '00' 
			WHEN extend9 IN ( 'STSP', 'STBL' ) THEN
			substr( t.extend5, 0, 2 ) || '0000' ELSE t.extend5 
		END xzqhdm,
	t.extend4 ywlx 
FROM
	wfprocessinst t 
WHERE
	extend9 IN ( 'XJZJ', 'SJZJ', 'STSP', 'STBL', 'SJBL', 'XJBL' ) 
	AND extend3 = 'JSYDSP' 
	AND currentstate = 2 
	AND t.extend5 IS NOT NULL 
	) t
	LEFT JOIN ( SELECT processinstid, activityinstname hjmc, starttime FROM wfactivityinst WHERE activitytype = 'manual' AND currentstate = 2 ) t1 ON t.processinstid = t1.processinstid 
WHERE
	xzqhdm IS NOT NULL 
	AND hjmc IS NOT NULL 
	and t.xmmc not like '%测试%'
GROUP BY
	xzqhdm,
	hjmc 
ORDER BY
	xzqhdm ASC,
	hjmc DESC
```

SELECT
	A.processinstid,
	A.ywid,
	A.ywlx,
	A.xmmc,
	to_char( A.DDSJ, 'yyyy-MM-dd HH24:mi:ss' ) DDSJ,
	to_char( C.endtime, 'yyyy-MM-dd HH24:mi:ss' ) LQSJ,
CASE
		
		WHEN A.currentstate = 7 THEN
		B.partiname ELSE NULL 
	END BJR,
CASE
	
	WHEN C.currentstate = 4 THEN
	'未领取' ELSE '已领取' 
	END LQZT,
	to_char( A.BJSJ, 'yyyy-MM-dd HH24:mi:ss' ) BJSJ,
	A.SHCS 
FROM
	(
	SELECT
		A.extend2 ywid,
		A.extend4 ywlx,
		A.processinstid,
		A.currentstate,
		A.extend1 XMMC,
		min( B.createtime ) ddsj,
		min( B.starttime ) lqsj,
		A.endtime bjsj,
		sum( sh ) shcs 
	FROM
		( SELECT * FROM tdglbps.wfprocessinst WHERE extend9 = 'JSYD_LSYD' ) A
		INNER JOIN (
		SELECT
			processinstid,
			createtime,
			starttime,
		CASE
				
				WHEN currentstate IN ( 12, 13 ) THEN
				1 ELSE 0 
			END sh 
	FROM
		tdglbps.wfworkitem 
	WHERE
		activitydefid = 'lsydsp_sh' 
	) B ON A.processinstid = B.processinstid 
GROUP BY
	A.extend1,
	A.processinstid,
	A.currentstate,
	A.endtime,
	A.extend2,
	A.extend4 
	) A
	LEFT JOIN (
	SELECT
		* 
	FROM
		(
		SELECT
			processinstid,
			participant,
			partiname,
			currentstate,
			endtime,
			row_number ( ) over ( partition BY processinstid ORDER BY starttime DESC ) rowIndex 
		FROM
			tdglbps.wfworkitem 
		WHERE
			activitydefid = 'lsydsp_sh' 
		) 
	WHERE
		rowIndex = 1 
	) B ON A.processinstid = B.processinstid
	LEFT JOIN (
	SELECT
		* 
	FROM
		(
		SELECT
			processinstid,
			participant,
			partiname,
			currentstate,
			endtime,
			row_number ( ) over ( partition BY processinstid ORDER BY starttime ASC ) rowIndex

			tdglbps.wfworkitem

			activitydefid = 'lsydsp_sh' 
		)
