-- 工程规划
SELECT xzqhds,xzqhds_mc,sum(bjaj) gcbjaj,sum(zbaj) gczbaj,sum(zbmj) zbmj from (
SELECT xzqhds,xzqhds_mc,case when CURRENTSTATE=7 then 1 end bjaj,case when CURRENTSTATE=2 then 1 end zbaj, zbmj from (
SELECT sjd.xmid,sjd.ywmc, YWXLMC,LCSLID,gcjbxx.xzqhds,gcjbxx.xzqhds_mc,CURRENTSTATE,zbmj from oa_sjd sjd  
<!--more-->
left join ghxk_gcjs_jbxxb gcjbxx on sjd.XMID=gcjbxx.xmid 
left join ghxk_xcjs_jbxxb xcjbxx on sjd.XMID=xcjbxx.xmid 
left join `bps_td`.`wfprocessinst` wf on wf.PROCESSINSTID=sjd.LCSLID 
left join (SELECT xmid,round(sum(DKMJ)/10000,4) zbmj FROM `txfx_dkxx`  group by xmid) dkxx on sjd.xmid=dkxx.xmid
where YWXLMC='xcjsghxk' and YWFLBM='GCGHXJBA' and ywmc<>'新建项目' and ywmc is not null and ywmc<>''  and (gcjbxx.xmid is not null or xcjbxx.xmid is not null) and gcjbxx.xmid  is not null 
# and ENDTIME>='2019'  
 group by sjd.xmid, sjd.ywmc,YWXLMC,LCSLID,gcjbxx.xzqhds,gcjbxx.xzqhds_mc,CURRENTSTATE
) v1 ) v1 group by xzqhds,xzqhds_mc
;


-- 乡村规划
SELECT xzqhds,xzqhds_mc,sum(bjaj) xcbjaj,sum(zbaj) xczbaj,sum(zbmj) zbmj from (
SELECT xzqhds,xzqhds_mc,case when CURRENTSTATE=7 then 1 end bjaj,case when CURRENTSTATE=2 then 1 end zbaj,zbmj from (
SELECT sjd.xmid, sjd.ywmc,YWXLMC,LCSLID,xcjbxx.xzqhds,xcjbxx.xzqhds_mc,CURRENTSTATE,zbmj from oa_sjd sjd  
left join ghxk_gcjs_jbxxb gcjbxx on sjd.XMID=gcjbxx.xmid 
left join ghxk_xcjs_jbxxb xcjbxx on sjd.XMID=xcjbxx.xmid 
left join `bps_td`.`wfprocessinst` wf on wf.PROCESSINSTID=sjd.LCSLID 
left join (SELECT xmid,round(sum(DKMJ)/10000,4) zbmj FROM `txfx_dkxx`  group by xmid) dkxx on sjd.xmid=dkxx.xmid
where YWXLMC='xcjsghxk' and YWFLBM='XCGHXJBA'  and ywmc<>'新建项目' and ywmc is not null and ywmc<>''  and (gcjbxx.xmid is not null or xcjbxx.xmid is not null) and xcjbxx.xmid  is not null
group by sjd.xmid, sjd.ywmc,YWXLMC,LCSLID,xcjbxx.xzqhds,xcjbxx.xzqhds_mc,CURRENTSTATE
) v1 ) v1 group by xzqhds,xzqhds_mc
;
