 select tt2.dlmc, tt1.* from
        (
            select dlbm,
                   round(sum(zymj_20)/10000,4) zymj_20,
                   round(sum(zymj_30)/10000,4) zymj_30,
                   round((sum(zymj_20)+sum(zymj_30))/10000,4) zymj  from
