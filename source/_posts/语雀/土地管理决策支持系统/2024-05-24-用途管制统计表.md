#### 农转用与土地征收（审批情况）含对应项目清单
##### 统计SQL（土地库）
```
--农转用与土地征收（审批情况）
WITH all_case AS(
SELECT * FROM jsyd_bpxx WHERE (PZWH IS NOT NULL OR PZRQ IS NOT NULL)   AND PZWH NOT LIKE '%〕号%' AND (YWXL='JSYDSP' OR YWXL='JSYDSP_BA') AND to_char(pzrq,'yyyy-mm-dd')>='2023-07-01' AND to_char(pzrq,'yyyy-mm-dd')<'2023-08-01' 
  )
SELECT * FROM (SELECT d.dictid ,d.dictname FROM jsyd_dic_entry d WHERE substr(d.dictid,5,2)='00' AND  d.dicttypeid='XZQHDM' AND substr(d.dictid,0,4)<>'4499') dic LEFT JOIN (
SELECT dictnamedsdm,dictnameds,COUNT(1),SUM(MJ),SUM(XZJSYDZB),SUM(nydzb),SUM(gdzb),SUM(gwysp),SUM(gwyspmj),SUM(szfsp),SUM(szfspmj),SUM(wtsxsp),SUM(wtsxspmj),SUM(sxsp),SUM(sxspmj) FROM (
SELECT t.YWID, XMMC, XMALX, XMND, DLND, QSND, XZQHDM, XZQHMC, SPJB,
( SELECT d.dictname FROM jsyd_dic_entry d WHERE d.dictid =substr(t.xzqhdm,0,4)||'00'  AND d.dicttypeid='XZQHDM') dictnameds,
( SELECT d.dictid FROM jsyd_dic_entry d WHERE d.dictid =substr(t.xzqhdm,0,4)||'00'  AND d.dicttypeid='XZQHDM') dictnamedsdm,
CASE WHEN spjb IN ('01','08') THEN 'gwysp'
  WHEN spjb IN ('09') THEN 'szfsp' 
    WHEN spjb IN ('03','06') THEN 'wtsxsp'
```
