```plsql
SELECT CASE 
         WHEN COUNT(1) > 0 THEN 
          '3' 
         ELSE 
          '0' 
       END jcjg, 
       CASE 
         WHEN COUNT(1) > 0 THEN 
          '规则通过' 
         ELSE 
          '必须在备案信息审核表中填写审批意见且审批意见为通过时才能提交' 
       END jcjgxx 
  FROM (SELECT CASE WHEN yjjl='通过' THEN '3' ELSE '0' END jcjg 
          FROM jsyd_yj a 
          LEFT JOIN tdglbps.wfworkitem b 
            ON a.gzxid = b.workitemid 
         WHERE ywid = ':ywid' 
           AND gzxid = ':workitemid')
 WHERE jcjg = 3
```
