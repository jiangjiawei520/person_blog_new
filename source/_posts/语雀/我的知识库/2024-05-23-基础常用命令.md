--一、表空间

--1、创建一个用户表空间

--这个语句创建了一个TBS的表空间，初始化两个文件TBS01.dbf，初始大小为128M 打开自动扩展，每次扩展4M，最大大小是10G

create tablespace TDGLYWK datafile 'D:\dmdbms\data\DAMENG\TDGLYWK.DBF' size 128 autoextend on next 4 maxsize 10240;

<!--more-->


--2、为表空间添加数据文件

--数据文件操作

alter tablespace TDGLYWK add datafile  'D:\dmdbms\data\DAMENG\TDGLYWK2.DBF' size 500;

alter tablespace TDGLYWK add datafile  'D:\dmdbms\data\DAMENG\TDGLYWK3.DBF' size 128 autoextend on next 4 maxsize 10240;





--3、删除表空间

--删除表空间只能删除没有使用过的表空间。

drop tablespace TDGLYWK;



--查询表空间有的数据文件

SELECT ts.NAME, df.PATH FROM V$TABLESPACE AS ts, V$DATAFILE AS df  WHERE ts.ID= df.GROUP_ID;





--4、修改表空间名称

alter tablespace TDGLYWK  rename to TDGLYWK_new ;



--5、修改表空间状态

--1)、离线表空间

alter tablespace to TDGLYWK offline;

--2)、上线表空间

alter tablespace to TDGLYWK online;





--6、修改表空间数据缓冲区

alter tablespace TDGLYWK cache="KEEP";



alter tablespace TDGLYWK cache="NORMAL";



--7、移动数据文件

--要移动数据文件，必须让表空间处于脱机状态offline

alter tablespace TDGLYWK offline;

alter tablespace TDGLYWK rename datafile  'D:\dmdbms\data\DAMENG\TDGLYWK2.DBF' to 'D:\dmdbms\data\DAMENG\TDGLYWK4.DBF';

alter tablespace TDGLYWK online;

--查询表空间有的数据文件

select file_name,TABLESPACE_NAME from dba_data_files;





--8、表空间的备份

backup tablespace TDGLYWK  backupset 'tablespace_TDGLYWK_bak_01';



--9、表空间恢复

--1)、创建一张表

create table dba_tables_test tablespace TDGLYWK  as select *  from dba_tables  ;

select table_name,owner from dba_tables_test limit 5;





--二、用户

--1、创建用户

--方式一：DM管理工具创建

--方式二：创建用户TDGLEOS 并指定默认的表空间及默认的索引表空间。

CREATE USER "TDGLEOS" IDENTIFIED BY "TDGLEOS123456" DEFAULT TABLESPACE "TDGLYWK" DEFAULT INDEX TABLESPACE "TDGLYWK";

GRANT DBA TO TDGLEOS;
