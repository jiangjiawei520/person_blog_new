```
CREATE DEFINER=`root`@`%` FUNCTION `GROUP_CONCAT_SUBSTRING_INDEX`(val VARCHAR(4000)) RETURNS varchar(4000) CHARSET utf8
BEGIN
	DECLARE result VARCHAR(4000) DEFAULT '';
SELECT 
	(
	SELECT GROUP_CONCAT(DISTINCT SUBSTRING_INDEX(SUBSTRING_INDEX(a.path,',',b.help_topic_id + 1),',',-1)) as path
FROM  
(SELECT val as path FROM dual) a
JOIN 
mysql.help_topic b 
ON b.help_topic_id < (LENGTH(a.path) - LENGTH(REPLACE(a.path,',','')) + 1)
	) path  INTO result;


	RETURN result;
END
```
