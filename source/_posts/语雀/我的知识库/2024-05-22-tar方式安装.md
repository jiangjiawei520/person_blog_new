# 1）下载安装包
https://downloads.mysql.com/archives/community/网站上选择对应的安装版本下载，这里以5.7.33的tar包为例
![image.png](https://cdn.nlark.com/yuque/0/2022/png/12484160/1655432129489-a367334c-26e9-45ca-93cb-a6dd3d4158bf.png#clientId=u7fd88fb1-b5c4-4&from=paste&id=u05f9d253&originHeight=886&originWidth=1594&originalType=url&ratio=1&rotation=0&showTitle=false&size=169668&status=done&style=none&taskId=ud781c786-711f-443b-abc8-a08e0d6a239&title=)
```bash
[root@mother-board tools]# wget https://downloads.mysql.com/archives/get/p/23/file/mysql-5.7.33-linux-glibc2.12-x86_64.tar.gz
[root@mother-board tools]# tar -zxvf mysql-5.7.33-linux-glibc2.12-x86_64.tar.gz
[root@mother-board tools]# mv mysql-5.7.33-linux-glibc2.12-x86_64 mysql
[root@mother-board tools]# mv mysql /usr/local/
```

## 2）创建mysql用户
首先删除系统已有的mysql
```bash
#删除mariadb
[root@mother-board ~]# rpm -qa|grep mariadb
mariadb-libs-5.5.68-1.el7.x86_64
[root@mother-board ~]# rpm -e --nodeps mariadb-libs-5.5.68-1.el7.x86_64
#和删除mariadb一样，列出了多少个，就一个个通过：rmp -e --nodeps来删除
[root@mother-board ~]# rpm -qa|grep mysql

```
创建一个不需要登录的mysql账号，给对应的目录授予权限。
```bash
# 创建不需要登录的一个mysql账号，启动MySQL服务时会使用该账号
[root@mother-board tools]# groupadd mysql
[root@mother-board tools]# useradd -r -g mysql -s /bin/false mysql
#与上面作用一样
#[root@mother-board tools]# useradd -r -g mysql -s /sbin/nologin mysql
#授予权限
[root@mother-board tools]# chown -R mysql:mysql /usr/local/mysql
[root@mother-board tools]# mkdir -p /data/mysql
[root@mother-board tools]# chown -R mysql:mysql /data/mysql

```

## 5）添加环境变量
```bash
#设置环境变量
[root@mother-board mysql]# vi /etc/profile
export MYSQL_HOME=/usr/local/mysql
export PATH=$PATH:${MYSQL_HOME}/bin
[root@mother-board mysql]# source /etc/profile

```
## 6）添加到系统服务
```bash
[root@mother-board mysql]# cd support-files
#修改mysql.server中的basedir、datadir、mysqld_pid_file_path对应的值
[root@mother-board support-files]# vi mysql.server
basedir=/usr/local/mysql
datadir=/data/mysql/data
mysqld_pid_file_path=/data/mysql/mysql.pid

#先启动一下mysql
#1、查看mysql环境变量
[root@mother-board mysql]# vi /etc/profile
export MYSQL_HOME=/usr/local/mysql
export PATH=$PATH:${MYSQL_HOME}/bin
#2、临时启动方式
${MYSQL_HOME}/support-files/mysql.server start
#3、在/usr/lib/systemd/system目录下增加mysql.service，内容如下，其中XXX.pid即为前面配置的pid文件路径
[Unit]
Description=Mysql
After=syslog.target network.target remote-fs.target nss-lookup.target
 
[Service]
#Type=forking
Type=simple
PIDFile=/data/mysql/mysql.pid
ExecStart=/usr/local/mysql/support-files/mysql.server start
ExecReload=/bin/kill -s HUP $MAINPID
ExecStop=/bin/kill -s QUIT $MAINPID
PrivateTmp=false
 
[Install]
WantedBy=multi-user.target

#4、重新载入单元，扫描新的或有变化的单元，使刚刚添加的mysql.service生效
systemctl daemon-reload

#5、通过如下命令操作Mysql，四条命令分别为：设为开机启动、启动、停止、重启
systemctl enable mysql
systemctl start mysql
systemctl stop mysql
systemctl restart mysql
```

## 11）常见问题
### 1045 - Access denied for user 'root'@'xxx'(using password:YES)
原因：mysql没有开启远程连接的权限
解决方案：
1、登录mysql命令框
2、查看数据库的host是否有你的远程ip或者有%（代表所有IP都可以连接）select host,user from user;
3、如果遇到ERROR 1046 (3D000): No database selected
1）、执行 use mysql 即可
4、如果没有你的ip或者%
则执行：update user set host = ‘%’ where user = ‘root’ and host = ‘localhost’;
5、执行flush privileges;即可

### MySQL8修改密码策略报错：ERROR 1193 (HY000): Unknown system variable ‘validate_password_policy‘
原因分析：
使用命令：
```sql
mysql> select plugin_name, plugin_status from information_schema.plugins where plugin_name like 'validate%';
1
或
mysql> SHOW VARIABLES LIKE 'validate_password%';
```
解决方案：
```sql
install plugin validate_password soname 'validate_password.so';
```
### New main PID * does not belong to service, and PID file is not owned by root
```sql
systemctl status mysql.service
Jun 13 07:11:08 0ca2216f67ad systemd[1]: New main PID 2268 does not belong to service, and PID file is not owned by root. Refusing

vi /usr/lib/systemd/system/mysql.service
#Type=forking
Type=simple
这样改就不报错了
```
