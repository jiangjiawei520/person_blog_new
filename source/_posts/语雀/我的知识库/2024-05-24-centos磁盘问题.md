```
du -ach *    #这个能看到当前目录下的所有文件占用磁盘大小和总大小
 
du -sh       #查看当前目录总大小
 
du -sh *     #查看所有子目录大小
 
<!--more-->
du -sh ./*   #查看当前目录下所有文件/文件夹的大小 
 
lsof | grep delete    #如果怀疑删掉的数据还在占用磁盘空间试试这个
 
kill -9 pid           #结束掉进程就能释放磁盘空间了
```
