## 服务启动、停止、重启
#### 全部配置的服务
```sql
sh service.sh start all
sh service.sh stop all
sh service.sh restart all
```
<!--more-->
#### 单个配置的服务
```sql
sh service.sh start 单个服务名称(如attach)
sh service.sh stop 单个服务名称(如attach)
sh service.sh restart 单个服务名称(如attach)
```
## 脚本
### 初始脚本service.sh
```sql
#!/bin/sh 
# 微服务服务启动脚本
# jiangjiawei
# 2021-5-20

# 定义需要启动的微服务
export attach=/usr/local/geo-deploy/geo-attach/geo-attach-1.0.2-SNAPSHOT.jar

# 定义需要启动的微服务端口,用于查找进程,关闭微服务
export attach_port=9025

# 定义需要启动的微服务日志存放文件
export attach_log=/usr/local/geo-deploy/geo-attach/geo-attach.out

# ---------------------------启动函数---------------------------------------
start_attach()
{
	echo "--------1、attach 开始启动--------------"
	nohup java -Xms512m -Xmx512m -Xss1M -jar $attach > $attach_log 2>&1 &
	attach_pid=`lsof -i:$attach_port|grep "LISTEN"|awk '{print $2}'`
	until [ -n "$attach_pid" ]
		do
		  attach_pid=`lsof -i:$attach_port|grep "LISTEN"|awk '{print $2}'`  
		done
	echo "attach pid is $attach_pid" 
	echo "--------1、attach 启动成功--------------"
}

# ---------------------------停止函数---------------------------------------
stop_attach()
{
	P_ID=`ps -ef | grep -w $attach | grep -v "grep" | awk '{print $2}'`
	if [ "$P_ID" == "" ]; then
		echo "===attach process not exists or stop success"
	else
		kill -9 $P_ID
		echo "attach killed success"
	fi 
}

case "$1" in 
start)  
	case "$2" in 
		attach|attach)
			## 1、启动attach
			start_attach 
		;;
		all|ALL)
		## 1、启动attach
			start_attach 
			echo "===startAll success==="
		;;
	esac
	;;  
 stop)
 case "$2" in 
		attach|attach)
			## 1、停止attach
			stop_attach 
		;;
		all|ALL)
			## 1、停止attach
			stop_attach 
			echo "===stopAll success==="
		;;
	esac
	;; 
restart)
case "$2" in 
		attach|attach)
			## 1、重启attach
			sh $0 stop attach
			sleep 2
			sh $0 start attach 
		;;
		all|ALL)
			sh $0 stop all
			sleep 10
			sh $0 start all
			echo "===restartAll success==="
		;;
	esac
	;;   
esac	
exit 0
```

# ---------------------------启动函数---------------------------------------
start_function()
{	
	printf "${YELLOW}"
	echo "--------1、微服务$2 开始启动--------------"

	# 判断文件夹是否存在
	if [ ! -d "$1" ]; then
		echo "$1文件夹不存在，即将退出！"
	else
		cd $1
		#判断文件是否存在
		if [ ! -e "$2" ]; then
			echo "微服务$2不存在,即将退出"
		else
			echo "微服务$2启动路径是：`pwd`"
			nohup java -Xms512m -Xmx512m -Xss1M -jar $2 > $3 2>&1 &
			pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`
			until [ -n "$pid" ]
				do
				  pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`  
				done
			echo " $2 微服务存放路径是 $1$2"
			echo " $2 日志文件是 $1$3"
			echo " $2 端口号是 $4"
			echo " $2 进程ID是 $pid"
			echo "--------2、微服务$2 启动成功--------------"
			printf "${RESET}"
		fi
	fi

}


# ---------------------------停止函数---------------------------------------
stop_function()
{
	printf "${YELLOW}"
	echo "--------1、微服务$2 开始停止--------------"
	P_ID=`ps -ef | grep -w $2 | grep -v "grep" | awk '{print $2}'`
	if [ "$P_ID" == "" ]; then
		echo "--------2、微服务$2 进程不存在或者已经停止--------------"
	else
		kill -9 $P_ID
		echo " $2 微服务存放路径是 $1$2"
		echo " $2 日志文件是 $1$3"
		echo " $2 端口号是 $4"
		echo " $2 停止进程ID是 $pid"
		echo "--------2、微服务$2 停止成功--------------"
	fi 
	printf "${RESET}"
}

# ---------------------------重启函数---------------------------------------
restart_function()
{
	stop_function $1 $2 $3 $4
	sleep 2
	start_function $1 $2 $3 $4
}

# ---------------------------启动函数---------------------------------------
start_function()
{	
	printf "${RED}"
	echo "--------启动函数步骤1、微服务$2 开始启动--------------"

	# 判断文件夹是否存在
	if [ ! -d "$1" ]; then
		echo "$1文件夹不存在，即将退出！"
	else
		cd $1
		#判断文件是否存在
		if [ ! -e "$2" ]; then
			echo "微服务$2不存在,即将退出"
		else
			echo "微服务$2启动路径是：`pwd`"
			nohup java $5 -jar  $2 >> $3 2>&1 &
			pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`
			until [ -n "$pid" ]
				do
				  pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`  
				done
			echo " $2 微服务启动命令是 nohup java $5 -jar  $2 >> $3 2>&1 &"
			echo " $2 微服务存放路径是 $1$2"
			echo " $2 日志文件是 $1$3"
			echo " $2 端口号是 $4"
			echo " $2 进程ID是 $pid"
			echo "--------启动函数步骤2、微服务$2 启动成功--------------"
			printf "${RESET}"
		fi
	fi

}


# ---------------------------停止函数---------------------------------------
stop_function()
{
	printf "${GREEN}"
	echo "--------停止函数步骤1、微服务$2 开始停止--------------"
	P_ID=`ps -ef | grep -w $2 | grep -v "grep" | awk '{print $2}'`
	if [ "$P_ID" == "" ]; then
		echo "--------停止函数步骤2、微服务$2 进程不存在或者已经停止--------------"
	else
		kill -9 $P_ID
		echo " $2 微服务存放路径是 $1$2"
		echo " $2 日志文件是 $1$3"
		echo " $2 端口号是 $4"
		echo " $2 停止进程ID是 $P_ID"
		echo "--------停止函数步骤2、微服务$2 停止成功--------------"
	fi 
	printf "${RESET}"
}

# ---------------------------重启函数---------------------------------------
restart_function()
{
	stop_function $1 $2 $3 $4 "$5"
	sleep 2
	start_function $1 $2 $3 $4 "$5"
}

# 需要传参五个传参 ftp的ip ftp的用户名 ftp的用户密码 需要下载到的路径地址 需要下载的微服务
installationPackage_function()
{
	printf "${PURPLE}"
echo "--------微服务下载函数步骤1、微服务$5 开始下载--------------"
echo "--------微服务下载函数步骤2、微服务$5下载路径为/usr/local/geo-deploy/geoInstallationPackage--------------"

ftp -n<<!
open $1
user $2 $3
binary
cd /usr/local/geo-deploy/geoInstallationPackage
lcd $4
prompt
get $5
close
bye
!
echo "--------微服务下载函数步骤4、微服务$5 下载完成--------------"
	printf "${RESET}"
}

# ---------------------------服务器登录函数---------------------------------------
login_function()
{	
	printf "${BLUE}"
	read -p "即将开始登录服务器,请输入需要登录的服务器ip：" ipadd
	read -p "即将开始登录服务器,请输入需要登录的用户名：" username
	printf "${RESET}"
	ssh $username@$ipadd
	printf "${BLUE}"
	echo "退出登录成功，当前服务器为：`hostname -I | awk -F " " '{printf $1}'`"
	printf "${RESET}"
}

# ---------------------------启动函数---------------------------------------
start_function()
{	
	printf "${RED}"
	echo "--------启动函数步骤1、微服务$2 开始启动--------------"

	# 判断文件夹是否存在
	if [ ! -d "$1" ]; then
		echo "$1文件夹不存在，即将退出！"
	else
		cd $1
		#判断文件是否存在
		if [ ! -e "$2" ]; then
			echo "微服务$2不存在,即将退出"
		else
			echo "微服务$2启动路径是：`pwd`"
			nohup java $5 -jar  $2 >> $3 2>&1 &
			pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`
			until [ -n "$pid" ]
				do
				  pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`  
				done
			echo " $2 微服务启动命令是 nohup java $5 -jar  $2 >> $3 2>&1 &"
			echo " $2 微服务存放路径是 $1$2"
			echo " $2 日志文件是 $1$3"
			echo " $2 端口号是 $4"
			echo " $2 进程ID是 $pid"
			echo "--------启动函数步骤2、微服务$2 启动成功--------------"
			echo "--------微服务更新返回最后十行结果--------------" 
			tail -n 10 $1$3
			printf "${RESET}"
		fi
	fi

}


# ---------------------------停止函数---------------------------------------
stop_function()
{
	printf "${GREEN}"
	echo "--------停止函数步骤1、微服务$2 开始停止--------------"
	P_ID=`ps -ef | grep -w $2 | grep -v "grep" | awk '{print $2}'`
	if [ "$P_ID" == "" ]; then
		echo "--------停止函数步骤2、微服务$2 进程不存在或者已经停止--------------"
	else
		kill -9 $P_ID
		echo " $2 微服务存放路径是 $1$2"
		echo " $2 日志文件是 $1$3"
		echo " $2 端口号是 $4"
		echo " $2 停止进程ID是 $P_ID"
		echo "--------停止函数步骤2、微服务$2 停止成功--------------"
	fi 
	printf "${RESET}"
}

# ---------------------------重启函数---------------------------------------
restart_function()
{
	stop_function $1 $2 $3 $4 "$5"
	sleep 2
	start_function $1 $2 $3 $4 "$5"
}

# ---------------------------微服务下载函数---------------------------------------
# 需要传参五个传参 ftp的ip ftp的用户名 ftp的用户密码 需要下载到的路径地址 需要下载的微服务
installationPackage_function()
{
	printf "${PURPLE}"
echo "--------微服务下载函数步骤1、微服务$5 开始下载--------------"
echo "--------微服务下载函数步骤2、微服务$5下载路径为/usr/local/geo-deploy/geoInstallationPackage--------------"
echo "--------微服务下载函数步骤3、微服务$5 下载后存储路径为$4--------------"
ftp -n<<!
open $1
user $2 $3
binary
cd /usr/local/geo-deploy/geoInstallationPackage
lcd $4
prompt
get $5
close
bye
!
echo "--------微服务下载函数步骤4、微服务$5 下载完成--------------"
	printf "${RESET}"
}

# ---------------------------微服务更新函数---------------------------------------

{
	printf "${CYAN}"
echo "--------微服务$2 更新开始--------------"

cd $1
mv $2 $2$YEARS$MONTH$DAY

printf "${CYAN}"
echo "--------微服务更新函数步骤4、开始重启微服务$2--------------"
restart_function $1 $2 $3 $4 "$5"
printf "${CYAN}"
echo "--------微服务更新函数步骤5、更新微服务$2完成--------------" 
echo "--------微服务更新返回最后十行结果--------------" 
tail -n 10 $1$3
printf "${RESET}"
}

# ---------------------------启动函数---------------------------------------
start_function()
{	
	printf "${RED}"
	echo "--------启动函数步骤1、微服务$2 开始启动--------------"

	# 判断文件夹是否存在
	if [ ! -d "$1" ]; then
		echo "$1文件夹不存在，即将退出！"
	else
		cd $1
		#判断文件是否存在
		if [ ! -e "$2" ]; then
			echo "微服务$2不存在,即将退出"
		else
			echo "微服务$2启动路径是：`pwd`"
			nohup java $5 -jar  $2 >> $3$YEARS$MONTH$DAY$HMS.out 2>&1 &
			pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`
			until [ -n "$pid" ]
				do
				  pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`  
				done
			echo " $2 微服务启动命令是 nohup java $5 -jar  $2 >> $3$YEARS$MONTH$DAY$HMS.out 2>&1 &"
			echo " $2 微服务存放路径是 $1$2"
			echo " $2 日志文件是 $1$3"
			echo " $2 端口号是 $4"
			echo " $2 进程ID是 $pid"
			echo "--------启动函数步骤2、微服务$2 启动成功--------------"
			echo "--------微服务更新返回最后十行结果--------------" 
			tail -n 10 $1$3$YEARS$MONTH$DAY$HMS.out
			printf "${RESET}"
		fi
	fi

}


# ---------------------------停止函数---------------------------------------
stop_function()
{
	printf "${GREEN}"
	echo "--------停止函数步骤1、微服务$2 开始停止--------------"
	P_ID=`ps -ef | grep -w $2 | grep -v "grep" | awk '{print $2}'`
	if [ "$P_ID" == "" ]; then
		echo "--------停止函数步骤2、微服务$2 进程不存在或者已经停止--------------"
	else
		kill -9 $P_ID
		echo " $2 微服务存放路径是 $1$2"
		echo " $2 日志文件存放路径是 $1$3最新日期.out"
		echo " $2 端口号是 $4"
		echo " $2 停止进程ID是 $P_ID"
		echo "--------停止函数步骤2、微服务$2 停止成功--------------"
	fi 
	printf "${RESET}"
}

# ---------------------------重启函数---------------------------------------
restart_function()
{
	stop_function $1 $2 $3 $4 "$5"
	sleep 2
	start_function $1 $2 $3 $4 "$5"
}
