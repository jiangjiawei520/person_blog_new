## 一、如何批量替换某字段的部分值，该字段其他部分保持不变
1、创建测试表；
create table test_str_[replace](https://www.baidu.com/s?wd=replace&tn=SE_PcZhidaonwhc_ngpagmjz&rsv_dl=gh_pc_zhidao)(id varchar2(20),value varchar2(20));

2、插入测试数据；
insert into test_str_[replace](https://www.baidu.com/s?wd=replace&tn=SE_PcZhidaonwhc_ngpagmjz&rsv_dl=gh_pc_zhidao) values('1','abcbcade');
insert into test_str_[replace](https://www.baidu.com/s?wd=replace&tn=SE_PcZhidaonwhc_ngpagmjz&rsv_dl=gh_pc_zhidao) values('2','aaabcbca');
insert into test_str_replace values('3','aabcaabcab');
commit;
<!--more-->

3、查询表中全量数据；select t.*， rowid from test_str_replace t;

4、编写语句，将value字段中'abc'替换成'123'，其他保持不变；
   select t.*, regexp_replace(value,'abc','123') cnt from test_str_replace t ;
