query.js
```bash
db.getCollection('GeoFile.chunks').aggregate([
   {
     $lookup:
       {
         from: "GeoFile.files",
<!--more-->
         localField: "files_id",
         foreignField: "_id",
         as: "inventory_docs"
       }
  },{
        '$group': {
            '_id': {
                _id: '$_id',
                files_id: '$files_id',
                n: '$n',
                chunkSize: '$inventory_docs.chunkSize',
                uploadDate: '$inventory_docs.uploadDate'
                
            },
             num_tutorial: {
                $sum: 1
            }
        }
    }
]).forEach(
    function(document) {
       print(document._id._id+"," + document._id.files_id+"," + document._id.n+"," + document._id.chunkSize+"," + document._id.uploadDate +"," + document.num_tutorial);
    }
)
```
