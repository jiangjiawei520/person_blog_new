SELECT

	a.TABLE_NAME,

	b.table_comment ,

    a.COLUMN_NAME ,

    a.COLUMN_COMMENT  
<!--more-->

FROM

INFORMATION_SCHEMA.COLUMNS a 

LEFT JOIN information_schema.tables b on a.TABLE_NAME = b.table_name

WHERE a.table_schema = 'tysb_gdbs'  and a.TABLE_NAME='zwxx_xmjbxx'

order by a.TABLE_NAME
