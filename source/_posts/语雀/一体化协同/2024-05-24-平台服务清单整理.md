filter {
   if[fields][flag]=="geo-log-gateway_access"{
        json {
           source => "message"
        }
        date {
           match => ["logTime", "yyyy-MM-dd HH:mm:ss.SSS"]
        }
   }
   if[fields][flag]=="geo-log-exchange-gateway_access"{
        json {
           source => "message"
        }
        date {
           match => ["logTime", "yyyy-MM-dd HH:mm:ss.SSS"]
        }
   }
}

output {
  elasticsearch {
    hosts => ["http://173.16.33.33:9200", "http://173.16.33.192:9200","http://173.16.33.146:9200"]
