--建立临时表（需要建立索引）存储需要处理数据

SELECT * from cjfk_temp;
insert into  cjfk_temp (ywid) SELECT ywid from (
SELECT ywid from ZWXX_TD_CJFK_LXXX
UNION all
SELECT ywid from ZWXX_TD_CJFK_KCSJXX
UNION all
SELECT ywid from ZWXX_TD_CJFK_SSXX
UNION all
SELECT ywid from ZWXX_TD_CJFK_JGYSXX
UNION all
SELECT ywid from ZWXX_TD_CJFK_YSQRXX) t1;
--备份
--处理数据
SELECT * from ZWXX_TD_CJFK_SGJDXX where ywid in (SELECT ywid from cjfk_temp) and data_identification='zrzyttdjc';
