# ---------------------------启动函数---------------------------------------
start_function()
{	
printf "${YELLOW}"
echo "--------1、微服务$2 开始启动--------------"
# 判断文件夹是否存在
if [ ! -d "$1" ]; then
echo "$1文件夹不存在，即将退出！"
exit 1
fi
cd $1
echo "微服务$2启动路径是：`pwd`"
nohup java -Xms512m -Xmx512m -Xss1M -jar $2 > $3 2>&1 &
pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`
until [ -n "$pid" ]
do
pid=`lsof -i:$4|grep "LISTEN"|awk '{print $2}'`  
done
echo " $2 微服务存放路径是 $1$2"
echo " $2 日志文件是 $1$3"
echo " $2 端口号是 $4"
echo " $2 进程ID是 $pid"
echo "--------2、微服务$2 启动成功--------------"
printf "${RESET}"
}


# ---------------------------停止函数---------------------------------------
stop_function()
{
printf "${YELLOW}"
echo "--------1、微服务$2 开始停止--------------"
P_ID=`ps -ef | grep -w $2 | grep -v "grep" | awk '{print $2}'`
if [ "$P_ID" == "" ]; then
echo "--------2、微服务$2 进程不存在或者已经停止--------------"
else
kill -9 $P_ID
echo " $2 微服务存放路径是 $1$2"
echo " $2 日志文件是 $1$3"
echo " $2 端口号是 $4"
echo " $2 停止进程ID是 $pid"
echo "--------2、微服务$2 停止成功--------------"
fi 
printf "${RESET}"
}

# ---------------------------重启函数---------------------------------------
restart_function()
{
stop_function $1 $2 $3 $4
sleep 2
start_function $1 $2 $3 $4
}
